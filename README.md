# cachy-arch-dwm
Post install script for CachyOS and vanilla Arch with my DWM setup (only for myself!)

./cachy-arch-dwm-installer.sh --dry-run

./cachy-arch-dwm-installer.sh --session-mode startx

./cachy-arch-dwm-installer.sh --verify --session-mode startx

./cachy-arch-dwm-installer.sh --session-mode sddm

./cachy-arch-dwm-installer.sh --verify --session-mode sddm
